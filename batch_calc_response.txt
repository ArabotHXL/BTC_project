        // Available miner models with specifications
        const minerModels = {
            'Antminer S19 Pro': { hashrate: 110, power: 3250 },
            'Antminer S19j Pro': { hashrate: 100, power: 3068 },
            'Antminer S21': { hashrate: 200, power: 3550 },
            'Antminer S19 XP': { hashrate: 140, power: 3010 },
            'Antminer T19': { hashrate: 84, power: 3150 },
            'WhatsMiner M50S': { hashrate: 126, power: 3276 },
            'WhatsMiner M53S': { hashrate: 226, power: 6554 },
            'WhatsMiner M56S': { hashrate: 212, power: 5550 },
            'AvalonMiner 1366': { hashrate: 100, power: 3420 },
            'AvalonMiner 1466': { hashrate: 150, power: 3420 }
        };

        let minerRowCount = 0;
        let calculationResults = [];
        const maxMiners = 100;
        let bypassUsed = false;

        // Initialize page
        document.addEventListener('DOMContentLoaded', function() {
            updateQuotaDisplay();
            updateEmptyState();
            
            // Load network data immediately and update price if real-time is enabled
            loadNetworkData().then(() => {
                // Ensure price is updated on initial load
                const useRealtime = document.getElementById('useRealTimeData');
                if (useRealtime && useRealtime.checked && networkDataCache && networkDataCache.btc_price) {
                    document.getElementById('globalBtcPrice').value = Math.round(networkDataCache.btc_price);
                }
            });
            
            // Auto-refresh network data every 30 seconds if real-time is enabled
            setInterval(() => {
                const useRealtime = document.getElementById('useRealTimeData');
                if (useRealtime && useRealtime.checked) {
                    loadNetworkData();
                }
            }, 30000);
        });

        // Update quota display in the header
        function updateQuotaDisplay() {
            const currentCountEl = document.getElementById('currentCount');
            const progressBarEl = document.getElementById('quotaProgress');
            
            if (currentCountEl) {
                currentCountEl.textContent = '0';
            }
            
            if (progressBarEl) {
                progressBarEl.style.width = '0%';
                progressBarEl.className = 'progress-bar bg-primary';
            }
        }

        function addMinerRow() {
            const tbody = document.getElementById('minersTableBody');
            const rowId = `miner-row-${++minerRowCount}`;
            
            const row = document.createElement('tr');
            row.id = rowId;
            row.innerHTML = `
                <td>
                    <select class="form-select form-select-sm" onchange="updateMinerSpecs(this, '${rowId}')">
                        <option value="">选择型号</option>
                        ${Object.keys(minerModels).map(model => 
                            `<option value="${model}">${model}</option>`
                        ).join('')}
                    </select>
                </td>
                <td>
                    <input type="number" class="form-control form-control-sm" 
                           min="1" max="1000" value="1" onchange="updateTotalCount()">
                </td>
                <td>
                    <input type="number" class="form-control form-control-sm" 
                           min="100" max="10000" placeholder="3250" step="10">
                </td>
                <td>
                    <input type="number" class="form-control form-control-sm" 
                           min="0.01" max="1" value="0.08" step="0.001">
                </td>
                <td>
                    <button class="btn btn-outline-danger btn-sm" onclick="removeMinerRow('${rowId}')">
                        <i class="bi bi-trash"></i>
                    </button>
                </td>
            `;
            
            tbody.appendChild(row);
            updateEmptyState();
            updateTotalCount();
        }

        function removeMinerRow(rowId) {
            document.getElementById(rowId).remove();
            updateEmptyState();
            updateTotalCount();
        }

        function updateMinerSpecs(selectElement, rowId) {
            const selectedModel = selectElement.value;
            if (selectedModel && minerModels[selectedModel]) {
                const row = document.getElementById(rowId);
                const powerInput = row.querySelector('td:nth-child(3) input');
                powerInput.value = minerModels[selectedModel].power;
            }
        }

        function updateTotalCount() {
            const rows = document.querySelectorAll('#minersTableBody tr');
            let totalMiners = 0;
            
            rows.forEach(row => {
                const quantityInput = row.querySelector('td:nth-child(2) input');
                if (quantityInput && quantityInput.value) {
                    totalMiners += parseInt(quantityInput.value) || 0;
                }
            });
            
            // Update quota display
            const currentCountSpan = document.getElementById('currentCount');
            const remainingCountSpan = document.getElementById('remainingCount');
            const progressBar = document.getElementById('quotaProgress');
            
            if (currentCountSpan && remainingCountSpan && progressBar) {
                currentCountSpan.textContent = totalMiners;
                remainingCountSpan.textContent = Math.max(0, maxMiners - totalMiners);
                
                const progressPercent = Math.min((totalMiners / maxMiners) * 100, 100);
                progressBar.style.width = progressPercent + '%';
                
                // Update progress bar color based on usage
                if (progressPercent >= 100) {
                    progressBar.className = 'progress-bar bg-danger';
                } else if (progressPercent >= 80) {
                    progressBar.className = 'progress-bar bg-warning';
                } else {
                    progressBar.className = 'progress-bar bg-success';
                }
            }
            
            // Update calculate button state
            updateCalculateButton(totalMiners);
        }

                if (totalMiners > maxMiners && !bypassUsed) {

                    calculateBtn.innerHTML = '<i class="bi bi-lock me-2"></i>需要升级';
                } else {

        function calculateBatchProfitability() {
            const miners = collectMinerData();
            
            if (miners.length === 0) {
                alert('请至少添加一台矿机再进行计算。');
                return;
            }

            const totalMiners = miners.reduce((sum, m) => sum + m.quantity, 0);
            
            // Check quota limits
            if (totalMiners > maxMiners && !bypassUsed) {
                showUpgradeModal(totalMiners);
                return;
            }

            // Show loading state
            showLoadingResults();

            // Prepare request data with safe defaults
            const btcPriceEl = document.getElementById('globalBtcPrice');
            const useRealtimeEl = document.getElementById('useRealTimeData');
            
            const requestData = {
                miners: miners,
                settings: {
                    btc_price: btcPriceEl ? btcPriceEl.value : null,
                    use_realtime: useRealtimeEl ? useRealtimeEl.checked : true
                }
            };

            // Send batch calculation request
            fetch('/api/batch-calculate', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(requestData)
            })
            .then(response => {
                console.log('Response status:', response.status);
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                return response.json();
            })
            .then(data => {
                console.log('Received data:', data);
                if (data.success) {
                    calculationResults = data.results;
                    displayBatchResults(data.results, data.summary);
                    showExportOptions();
                } else {
                    if (data.error === 'upgrade_required') {
                        showUpgradeModal(totalMiners);
                    } else {
                        showErrorResults(data.message || 'Unknown error');
                    }
                }
            })
            .catch(error => {
                console.error('Calculation error:', error);
                console.error('Error details:', error.message);
                showErrorResults('计算失败，请重试。');
            });
        }

        function collectMinerData() {
            const miners = [];
            const rows = document.querySelectorAll('#minersTableBody tr');
            
            rows.forEach(row => {
                const cells = row.querySelectorAll('td');
                const model = cells[0].querySelector('select').value;
                const quantity = parseInt(cells[1].querySelector('input').value) || 0;
                const power = parseFloat(cells[2].querySelector('input').value) || 0;
                const electricity = parseFloat(cells[3].querySelector('input').value) || 0;
                
                if (model && quantity > 0 && power > 0 && electricity > 0) {
                    miners.push({
                        model: model,
                        quantity: quantity,
                        power_consumption: power,  // Fixed: use correct field name
                        electricity_cost: electricity,
                        hashrate: minerModels[model]?.hashrate || 0
                    });
                }
            });
            
            return miners;
        }

        // CSV file processing
        function processCsvFile(event) {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(e) {
                const csv = e.target.result;
                const lines = csv.split('\n');
                
                // Clear existing miners
                clearAllMiners();
                
                // Skip header line
                for (let i = 1; i < lines.length; i++) {
                    const line = lines[i].trim();
                    if (!line) continue;
                    
                    const [model, quantity, power, electricity] = line.split(',').map(s => s.trim());
                    
                    if (model && quantity && power && electricity) {
                        addMinerRowFromData(model, quantity, power, electricity);
                    }
                }
                
                updateTotalCount();
            };
            
            reader.readAsText(file);
        }

        // Network data loading
        let networkDataCache = null;
        let lastNetworkDataUpdate = 0;
        const CACHE_DURATION = 30000; // 30 seconds

        async function loadNetworkData() {
            const now = Date.now();
            
            // Use cache if data is fresh
            if (networkDataCache && (now - lastNetworkDataUpdate) < CACHE_DURATION) {
                updateNetworkDisplay(networkDataCache);
                return;
            }

            try {
                const response = await fetch('/api/analytics-data');
                if (response.ok) {
                    const data = await response.json();
                    if (data.success && data.data) {
                        networkDataCache = data.data;
                        lastNetworkDataUpdate = now;
                        updateNetworkDisplay(data.data);
                        
                        // Update BTC price input if real-time is enabled
                        const useRealtime = document.getElementById('useRealTimeData');
                        if (useRealtime && useRealtime.checked && data.data.btc_price) {
                            document.getElementById('globalBtcPrice').value = Math.round(data.data.btc_price);
                        }
                    }
                }
            } catch (error) {
                console.warn('Failed to load network data:', error);
            }
        }

        function updateNetworkDisplay(data) {
            // Update network status display
            if (data.btc_price) {
                document.getElementById('currentBtcPrice').textContent = `$${data.btc_price.toLocaleString()}`;
            }
            if (data.network_difficulty) {
                const difficultyT = (data.network_difficulty / 1e12).toFixed(3);
                document.getElementById('currentDifficulty').textContent = `${difficultyT}T`;
            }
            if (data.network_hashrate) {
                document.getElementById('currentHashrate').textContent = `${data.network_hashrate} EH/s`;
            }
            
            // Update timestamp
            const now = new Date();
            const timeStr = now.toLocaleTimeString();
            document.getElementById('dataUpdateTime').textContent = `(Updated: ${timeStr})`;
        }
