        async function verifyUser(userId) {
            if (confirm('确认验证此用户的邮箱？')) {
                try {
                    const response = await fetch(`/admin/verify-user/${userId}`, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                        }
                    });
                    
                    if (response.ok) {
                        location.reload();
                    } else {
                        alert('操作失败，请重试');
                    }
                } catch (error) {
                    alert('网络错误，请重试');
                }
            }
        }

        async function extendAccess(userId) {
            const days = prompt('延长访问天数:', '30');
            if (days && !isNaN(days) && parseInt(days) > 0) {
                try {
                    const response = await fetch(`/admin/extend-access/${userId}`, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        body: JSON.stringify({days: parseInt(days)})
                    });
                    
                    if (response.ok) {
                        location.reload();
                    } else {
                        alert('操作失败，请重试');
                    }
                } catch (error) {
                    alert('网络错误，请重试');
                }
            }
        }
